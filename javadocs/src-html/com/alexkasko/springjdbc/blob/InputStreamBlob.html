001    package com.alexkasko.springjdbc.blob;
002    
003    import java.io.Closeable;
004    import java.io.InputStream;
005    import java.io.OutputStream;
006    
007    /**
008     * Interface for "readable" BLOBs
009     *
010     * @author alexkasko
011     * Date: 11/27/12
012     */
013    public interface InputStreamBlob extends Closeable {
014    
015        /**
016         * Returns BLOB unique ID from database
017         *
018         * @return BLOB ID
019         */
020        long getId();
021    
022        /**
023         * BLOB data stream
024         *
025         * @return BLOB data stream
026         */
027        InputStream inputStream();
028    
029        /**
030         * Shortcut method, reads all BLOB data into provided stream and close BLOB
031         *
032         * @param out stream to write data into
033         * @return number ob bytes read
034         * @throws BlobException on copy exception
035         */
036        long readAndClose(OutputStream out);
037    
038        /**
039         * Closes BLOB releasing database resources connected to it (if any)
040         * Must not throw any exception.
041         */
042        @Override
043        void close();
044    }
