001    package com.alexkasko.springjdbc.blob;
002    
003    /**
004     * Front-end for BLOB operations.
005     *
006     * @author alexkasko
007     * Date: 4/13/12
008     */
009    public interface BlobTool {
010        /**
011         * Creates BLOB and returns object to write data into
012         *
013         * @return BLOB stream to write to, it must be closed by the caller
014         * @throws BlobException on BLOB error
015         */
016        OutputStreamBlob create() throws BlobException;
017    
018        /**
019         * Loads BLOB from database
020         *
021         * @param id BLOB ID
022         * @return BLOB data stream, must be closed by the caller
023         * @throws BlobException on BLOB error
024         */
025        InputStreamBlob load(long id) throws BlobException;
026    
027        /**
028         * Loads BLOB from database in "detached" mode - copies compressed BLOB data
029         * into temporary file and closes database BLOB
030         *
031         * @param id BLOB ID
032         * @return BLOB data in detached (temp-file) mode
033         * @throws BlobException on BLOB error
034         */
035        DetachedBlob detach(long id) throws BlobException;
036    
037        /**
038         * Deletes BLOB from datbase
039         *
040         * @param id BLOB ID
041         * @throws BlobException on BLOB error
042         */
043        void delete(long id) throws BlobException;
044    }
