001    package com.alexkasko.springjdbc.blob;
002    
003    import com.alexkasko.springjdbc.compress.Compressor;
004    import org.springframework.jdbc.core.namedparam.MapSqlParameterSource;
005    import org.springframework.jdbc.core.namedparam.NamedParameterJdbcTemplate;
006    
007    import javax.sql.DataSource;
008    import java.sql.*;
009    
010    import static org.springframework.util.StringUtils.hasText;
011    
012    /**
013     * Basic implementation of Postgres-flavoured BLOB operations for other databases using standard JDBC API.
014     * Implementations may be true server-side (Oracle) or client-side with temp file.
015     * To simulate Postgres BLOBs {@code blob_storage_id_seq} sequence and
016     * {@code blob_storage(id bigint primary key, data blob)} must be created beforehand.
017     *
018     * @author alexkasko
019     * Date: 4/13/12
020     * @see BlobTool
021     */
022    
023    public abstract class AbstractJdbcBlobTool implements BlobTool {
024        protected final DataSource dataSource;
025        protected final NamedParameterJdbcTemplate jt;
026        protected final Compressor compressor;
027        protected final String generateIdSQL;
028        protected final String insertSQL;
029        protected final String loadSQL;
030        protected final String deleteSQL;
031    
032        /**
033         * Shortcut constructor with default SQL
034         *
035         * @param dataSource data source
036         * @param compressor compressor
037         */
038        public AbstractJdbcBlobTool(DataSource dataSource, Compressor compressor) {
039            this(dataSource, compressor,
040                    "select nextval('blob_storage_id_seq')",
041                    "insert into blob_storage(id, data) values(:id, :data)",
042                    "select data from blob_storage where id = :id",
043                    "delete from blob_storage where id = :id");
044        }
045    
046        /**
047         * @param dataSource data source
048         * @param compressor compressor
049         * @param generateIdSQL SQL to generate BLOB ID, default: {@code select nextval('blob_storage_id_seq')}
050         * @param insertSQL SQL to insert blob into database, default: {@code insert into blob_storage(id, data) values(:id, :data)}
051         * @param loadSQL SQL to load blob data from database, default: {@code select data from blob_storage where id = :id}
052         * @param deleteSQL SQL to delete BLOB from database, default: {@code delete from blob_storage where id = :id}
053         */
054        public AbstractJdbcBlobTool(DataSource dataSource, Compressor compressor, String generateIdSQL, String insertSQL,
055                                    String loadSQL, String deleteSQL) {
056            if(null == dataSource) throw new BlobException("Provided dataSource is null");
057            if(null == compressor) throw new BlobException("Provided compressor is null");
058            if(!hasText(generateIdSQL)) throw new BlobException("Provided generateIdSQL is blank");
059            if(!hasText(insertSQL)) throw new BlobException("Provided insertSQL is blank");
060            if(!hasText(loadSQL)) throw new BlobException("Provided loadSQL is blank");
061            if(!hasText(deleteSQL)) throw new BlobException("Provided deleteSQL is blank");
062            this.dataSource = dataSource;
063            this.jt = new NamedParameterJdbcTemplate(dataSource);
064            this.compressor = compressor;
065            this.generateIdSQL = generateIdSQL;
066            this.insertSQL = insertSQL;
067            this.loadSQL = loadSQL;
068            this.deleteSQL = deleteSQL;
069        }
070    
071        /**
072         * {@inheritDoc}
073         */
074        @Override
075        public InputStreamBlob load(long id) throws BlobException {
076            try {
077                MapSqlParameterSource params = new MapSqlParameterSource();
078                params.addValue("id", id, Types.BIGINT);
079                Blob loaded = jt.queryForObject(loadSQL, params, Blob.class);
080                if (null == loaded) throw new BlobException("No blob found for id: [" + id + "]");
081                return new ServerSideInputStreamBlob(id, loaded.getBinaryStream(), compressor);
082            } catch (Exception e) {
083                throw e instanceof BlobException ? (BlobException) e : new BlobException("Cannot load blob, id: [" + id + "]", e);
084            }
085        }
086    
087        /**
088         * {@inheritDoc}
089         */
090        @Override
091        public DetachedBlob detach(long id) throws BlobException {
092            try {
093                MapSqlParameterSource params = new MapSqlParameterSource();
094                params.addValue("id", id, Types.BIGINT);
095                Blob loaded = jt.queryForObject(loadSQL, params, Blob.class);
096                if (null == loaded) throw new BlobException("No blob found for id: [" + id + "]");
097                return new TempFileDetachedBlob(id, loaded.getBinaryStream(), compressor);
098            } catch (Exception e) {
099                throw e instanceof BlobException ? (BlobException) e : new BlobException("Cannot detach blob, id: [" + id + "]", e);
100            }
101        }
102    
103        /**
104         * {@inheritDoc}
105         */
106        @Override
107        public void delete(long id) throws BlobException {
108            try {
109                MapSqlParameterSource params = new MapSqlParameterSource();
110                params.addValue("id", id, Types.BIGINT);
111                int count = jt.update(deleteSQL, params);
112                if (1 != count) throw new BlobException("One row must have been deleted, but was: [" + count + "] " +
113                        "for blob, id: [" + id + "]");
114            } catch (Exception e) {
115                throw e instanceof BlobException ? (BlobException) e : new BlobException("Cannot delete blob, id: [" + id + "]", e);
116            }
117        }
118    }
