001    package com.alexkasko.springjdbc.blob;
002    
003    import com.alexkasko.springjdbc.compress.Compressor;
004    import org.springframework.jdbc.core.PreparedStatementCreator;
005    import org.springframework.jdbc.core.namedparam.MapSqlParameterSource;
006    import org.springframework.jdbc.core.namedparam.NamedParameterUtils;
007    import org.springframework.jdbc.core.namedparam.ParsedSql;
008    
009    import javax.sql.DataSource;
010    import java.io.InputStream;
011    import java.sql.Connection;
012    import java.sql.PreparedStatement;
013    import java.sql.SQLException;
014    import java.sql.Types;
015    
016    /**
017     * Client-side implementation of BLOB tool for databases, that don't support
018     * {@code java.io.OutputStream java.sql.Blob#setBinaryStream(long pos)} method properly
019     * (implement it on client in-memory or in similar fashion with temp file).
020     * This implementation creates BLOB as compressed temp file and transfer data from it after BLOB closed by client.
021     * Data is inserted from file into database using JDBC's streaming
022     * {@code void java.sql.PreparedStatement#setBinaryStream(int parameterIndex, java.io.InputStream x, long length)}
023     * method. So file data never goes fully into memory outside of JDBC driver,
024     * but may go into memory inside some creepy JDBC implementation.
025     * Not well suited for highload applications, may be used with H2 database to simulate PostgreSQL-like BLOBs in tests.
026     *
027     * @author alexkasko
028     * Date: 4/27/12
029     */
030    public class TempFileJdbcBlobTool extends AbstractJdbcBlobTool {
031    
032        /**
033         * Shortcut constructor with default SQL
034         *
035         * @param dataSource data source
036         * @param compressor compressor
037         */
038        public TempFileJdbcBlobTool(DataSource dataSource, Compressor compressor) {
039            super(dataSource, compressor);
040        }
041    
042        /**
043         * @param dataSource    data source
044         * @param compressor    compressor
045         * @param generateIdSQL SQL to generate BLOB ID, default: {@code select nextval('blob_storage_id_seq')}
046         * @param insertSQL     SQL to insert blob into database, default: {@code insert into blob_storage(id, data) values(:id, :data)}
047         * @param loadSQL       SQL to load blob data from database, default: {@code select data from blob_storage where id = :id}
048         * @param deleteSQL     SQL to delete BLOB from database, default: {@code delete from blob_storage where id = :id}
049         */
050        public TempFileJdbcBlobTool(DataSource dataSource, Compressor compressor, String generateIdSQL, String insertSQL, String loadSQL, String deleteSQL) {
051            super(dataSource, compressor, generateIdSQL, insertSQL, loadSQL, deleteSQL);
052        }
053    
054        /**
055         * {@inheritDoc}
056         */
057        @Override
058        public OutputStreamBlob create() throws BlobException {
059            try {
060                long id = jt.getJdbcOperations().queryForLong(generateIdSQL);
061                return new TempFileOutputStreamBlob(id, compressor, new Persister());
062            } catch (Exception e) {
063                throw e instanceof BlobException ? (BlobException) e : new BlobException("Cannot create blob", e);
064            }
065        }
066    
067        private class Persister implements OnClosePersister {
068    
069            @Override
070            public void persist(long id, InputStream data, long dataLength) {
071                jt.getJdbcOperations().update(new InsertPS(id, data, dataLength));
072            }
073    
074            private class InsertPS implements PreparedStatementCreator {
075                private final long id;
076                private final InputStream data;
077                private final long length;
078    
079                private InsertPS(long id, InputStream data, long length) {
080                    this.id = id;
081                    this.data = data;
082                    this.length = length;
083                }
084    
085                @Override
086                public PreparedStatement createPreparedStatement(Connection con) throws SQLException {
087                    // substitute named params, see NamedParameterJdbcTemplate#getPreparedStatementCreator(String sql, SqlParameterSource paramSource)
088                    ParsedSql parsedSql = NamedParameterUtils.parseSqlStatement(insertSQL);
089                    MapSqlParameterSource paramSource = new MapSqlParameterSource();
090                    paramSource.addValue("id", id, Types.BIGINT);
091                    paramSource.addValue("data", data, Types.BLOB);
092                    String sqlToUse = NamedParameterUtils.substituteNamedParameters(parsedSql, paramSource);
093                            Object[] params = NamedParameterUtils.buildValueArray(parsedSql, paramSource, null);
094                    PreparedStatement stmt = con.prepareStatement(sqlToUse);
095                    stmt.setLong(1, (Long) params[0]);
096                    if(length < Integer.MAX_VALUE) {
097                        stmt.setBinaryStream(2, (InputStream) params[1], (int) length);
098                    } else {
099                        //warn: JTDS doesn't support long sized blobs
100                        stmt.setBinaryStream(2, (InputStream) params[1], length);
101                    }
102                    return stmt;
103                }
104            }
105        }
106    }
