001    package com.alexkasko.springjdbc.blob;
002    
003    import com.alexkasko.springjdbc.compress.Compressor;
004    import org.springframework.jdbc.core.namedparam.MapSqlParameterSource;
005    import org.springframework.jdbc.datasource.DataSourceUtils;
006    
007    import javax.sql.DataSource;
008    import java.io.OutputStream;
009    import java.sql.Blob;
010    import java.sql.Connection;
011    import java.sql.Types;
012    
013    /**
014     * JDBC BLOB implementation that uses {@code java.io.OutputStream java.sql.Blob#setBinaryStream(long pos)}
015     * method to get output stream to remote BLOB. Among popular RDBMSes this method implemented properly (true server-side)
016     * only in Oracle (Postgres seems to has it too, but we already use native LO API for it).
017     *
018     * @author alexkasko
019     * Date: 4/27/12
020     */
021    public class ServerSideJdbcBlobTool extends AbstractJdbcBlobTool {
022    
023        /**
024         * Shortcut constructor with default SQL
025         *
026         * @param dataSource data source
027         * @param compressor compressor
028         */
029        public ServerSideJdbcBlobTool(DataSource dataSource, Compressor compressor) {
030            super(dataSource, compressor);
031        }
032    
033        /**
034         * @param dataSource    data source
035         * @param compressor    compressor
036         * @param generateIdSQL SQL to generate BLOB ID, default: {@code select nextval('blob_storage_id_seq')}
037         * @param insertSQL     SQL to insert blob into database, default: {@code insert into blob_storage(id, data) values(:id, :data)}
038         * @param loadSQL       SQL to load blob data from database, default: {@code select data from blob_storage where id = :id}
039         * @param deleteSQL     SQL to delete BLOB from database, default: {@code delete from blob_storage where id = :id}
040         */
041        public ServerSideJdbcBlobTool(DataSource dataSource, Compressor compressor, String generateIdSQL, String insertSQL, String loadSQL, String deleteSQL) {
042            super(dataSource, compressor, generateIdSQL, insertSQL, loadSQL, deleteSQL);
043        }
044    
045        /**
046         * {@inheritDoc}
047         */
048        @Override
049        public OutputStreamBlob create() throws BlobException {
050            try {
051                long id = jt.getJdbcOperations().queryForLong(generateIdSQL);
052                Connection conn = DataSourceUtils.doGetConnection(dataSource);
053                Blob blob = conn.createBlob();
054                MapSqlParameterSource params = new MapSqlParameterSource();
055                params.addValue("id", id, Types.BIGINT);
056                params.addValue("data", blob, Types.BLOB);
057                jt.update(insertSQL, params);
058                Blob created = jt.queryForObject(loadSQL, params, Blob.class);
059                OutputStream blobStream = created.setBinaryStream(1);
060                return new ServerSideOutputStreamBlob(id, blobStream, compressor);
061            } catch (Exception e) {
062                throw e instanceof BlobException ? (BlobException) e : new BlobException("Cannot create blob", e);
063            }
064        }
065    }
