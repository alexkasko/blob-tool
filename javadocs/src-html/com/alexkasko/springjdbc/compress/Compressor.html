001    package com.alexkasko.springjdbc.compress;
002    
003    import java.io.InputStream;
004    import java.io.OutputStream;
005    
006    /**
007     * Interface for different BLOB compression methods
008     *
009     * @author  alexkasko
010     * Date: 4/14/12
011     * @see com.alexkasko.springjdbc.blob.BlobTool
012     */
013    public interface Compressor {
014    
015        /**
016         * Must compress provided stream
017         *
018         * @param out stream to compress
019         * @return compressed stream
020         * @throws com.alexkasko.springjdbc.blob.BlobException on any exception
021         */
022        OutputStream wrapCompress(OutputStream out);
023    
024        /**
025         * Must decompress provided stream
026         *
027         * @param in stream to decompress
028         * @return decompressed stream
029         * @throws com.alexkasko.springjdbc.blob.BlobException on any exception
030         */
031        InputStream wrapDecompress(InputStream in);
032    }
