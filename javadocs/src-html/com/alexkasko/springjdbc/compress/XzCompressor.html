001    package com.alexkasko.springjdbc.compress;
002    
003    import org.tukaani.xz.LZMA2Options;
004    import org.tukaani.xz.XZInputStream;
005    import org.tukaani.xz.XZOutputStream;
006    
007    import java.io.IOException;
008    import java.io.InputStream;
009    import java.io.OutputStream;
010    
011    /**
012     * BLOB compressor implementation, uses high ratio <a href="http://tukaani.org/xz/">XZ</a> compression method
013     *
014     * @author alexkasko
015     * Date: 4/14/12
016     */
017    public class XzCompressor extends AbstractCompressor {
018        private final int level;
019    
020        /**
021         * No-arg constructor, uses {@code level=3}
022         */
023        public XzCompressor() {
024            this(3);
025        }
026    
027        /**
028         * Constructor, allows to set compression level
029         *
030         * @param level LZMA2 compression level
031         */
032        public XzCompressor(int level) {
033            this.level = level;
034        }
035    
036        /**
037         * {@inheritDoc}
038         */
039        @Override
040        protected OutputStream wrapCompressInternal(OutputStream out) throws IOException {
041            return new XZOutputStream(out, new LZMA2Options(level));
042        }
043    
044        /**
045         * {@inheritDoc}
046         */
047        @Override
048        protected InputStream wrapDecompressInternal(InputStream in) throws IOException {
049            return new XZInputStream(in);
050        }
051    }
